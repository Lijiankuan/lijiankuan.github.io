}

void MyString::print() {
    if (p_data_ == nullptr) {
        return;
    }
    printf("%s\n", p_data_);
}

void MyString::swap(MyString &str) {
    auto tmp = p_data_;
    p_data_ = str.p_data_;
    str.p_data_ = tmp;
}

MyString &MyString::operator=(const MyString &str) {
    if (this == &str) {
        return *this;
    }

    MyString backup(str);
    swap(backup);

    return *this;
}

    03：数组中重复的数字

    • 由于所有数字都是在[0, n-1]范围中的
    • 且给定数组长度为n
    • 那么数组下标可以充当哈希索引值
    • 通过循环遍历，将每个元素存储到对应的bucket中（此时哈希位置和数组在同一空间

    --

    • 循环遍历数组，遍历到索引i时，当前值为m
    • 若m==i，那么m存到了正确的 bucket
    • 若m!=i，那么不断地将m存到正确的 bucket
      • 若 bucket 已经存储了m，那么找到了重复的数字，方法返回 true
    class Solution {
public:
    // Parameters:
    //        numbers:     an array of integers
    //        length:      the length of array numbers
    //        duplication: (Output) the duplicated number in the array number
    // Return value:       true if the input is valid, and there are some duplications in the array number
    //                     otherwise false
    bool duplicate(int numbers[], int length, int* duplication) {
        if(numbers == nullptr || length <= 0){
            return false;
        }

        for(int i = 0; i < length; i++){
            while(numbers[i] != i){
                if(numbers[i] == numbers[numbers[i]]){
                    *duplication = numbers[i];
                    return true;
                }
                else{
                    int temp = numbers[numbers[i]];
                    numbers[numbers[i]] = numbers[i];
                    numbers[i] = temp;
                }
            }
        }

        return false;
    }
};

    class Solution {
public:
    bool Find(int target, vector<vector<int> > array) {
        int rows = array.size();
        int cols = rows == 0 ? 0 : array[0].size();

        int row = 0;
        int col = cols-1;

        while(row < rows && col >= 0){
            if(array[row][col] == target){
                return true;
            }
            if(array[row][col] < target){
                row++;
            }
            else{
                col--;
            }
        }

        return false;
    }
};

    05：替换空格

    class Solution {
public:
    void replaceSpace(char *str,int length) {
        if(str == nullptr || length <= 0){
            return ;
        }

        int countSpace = 0;
        for(int i = 0; i < length; i++){
            if(str[i] == ' '){
                countSpace++;
            }
        }

        int newLength = length + 2*countSpace;
        str[newLength] = '\0';

        int index = newLength;
        for(int i = length; i >= 0; i--){
            if(str[i] != ' '){
                str[index--] = str[i];
            }
            else{
                str[index--] = '0';
                str[index--] = '2';
                str[index--] = '%';
            }
        }
    }
};

    06：从尾到头打印链表

    /**
*  struct ListNode {
*        int val;
*        struct ListNode *next;
*        ListNode(int x) :
*              val(x), next(NULL) {
*        }
*  };
*/
class Solution {
public:
    vector<int> printListFromTailToHead(ListNode* head) {
        std::vector<int> vec;

        if(head == nullptr)
            return vec;

        std::stack<int> stk;
        ListNode* pNode = head;
        while(pNode != nullptr){
            stk.push(pNode->val);
            pNode = pNode->next;
        }

        while(!stk.empty()){
            vec.push_back(stk.top());
            stk.pop();
        }
        return vec;
    }
};

    07：重建二叉树

    /**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* reConstructBinaryTree(vector<int> pre,vector<int> vin) {
        if(pre.size() == 0){
            return nullptr;
        }

        return reConstructBinaryTreeCore(pre, vin, 0, pre.size()-1, 0, vin.size()-1);
    }

    TreeNode* reConstructBinaryTreeCore(vector<int>& pre,vector<int>& vin, int preStart, int preEnd, int inStart, int inEnd) {
        if(preStart > preEnd || inStart > inEnd)
            return nullptr;

        int valRoot = pre[preStart];
        TreeNode* rootNode = new TreeNode(valRoot);

        if(preStart == preEnd)
            return rootNode;

        int inOrderRootIndex = -1;
        for(int i = inStart; i <= inEnd; i++){
            if(vin[i] == valRoot){
                inOrderRootIndex = i;
                break;
            }
        }

        int leftTreeCount = inOrderRootIndex - inStart;
        int rightTreeCount = inEnd - inOrderRootIndex;

        rootNode->left = reConstructBinaryTreeCore(pre, vin, preStart+1, preStart+leftTreeCount, inStart, inOrderRootIndex-1);
        rootNode->right = reConstructBinaryTreeCore(pre, vin, preStart+leftTreeCount+1, preEnd, inOrderRootIndex+1, inEnd);

        return rootNode;
    }
};

    08：二叉树的下一节点

    /*
struct TreeLinkNode {
    int val;
    struct TreeLinkNode *left;
    struct TreeLinkNode *right;
    struct TreeLinkNode *next;
    TreeLinkNode(int x) :val(x), left(NULL), right(NULL), next(NULL) {

    }
};
*/
class Solution {
public:
    TreeLinkNode* GetNext(TreeLinkNode* pNode)
    {
        if(pNode == nullptr)
            return nullptr;
        if(pNode->right != nullptr){
            TreeLinkNode* pCur = pNode->right;
            while(pCur->left != nullptr){
                pCur = pCur->left;
            }
            return pCur;
        }
        else if(pNode->next != nullptr && pNode->next->left == pNode){
            return pNode->next;
        }
        else if(pNode->next != nullptr && pNode->next->right == pNode){
            TreeLinkNode* pCur = pNode->next;
            while(pNode->next != nullptr && pNode->next->right == pNode){
                pNode = pNode->next;
            }
            return pNode->next;
        }
        else{
            return nullptr;
        }
    }
};

    11：旋转数组的最小数字

    class Solution {
public:
    int minNumberInRotateArray(vector<int> rotateArray) {
        // 旋转0
        // 旋转非0
        // 数组长度1

        int pStart = 0;
        int pEnd = rotateArray.size()-1;

        while(pStart < pEnd){
            if(pEnd-pStart == 1){
                break;
            }
            int pMid = (pStart+pEnd)/2;
            if(rotateArray[pStart] == rotateArray[pEnd] && rotateArray[pStart] == rotateArray[pMid]){
                return searchMin(rotateArray);
            }
            if(rotateArray[pStart] <= rotateArray[pMid]){
                pStart = pMid;
            }
            else if(rotateArray[pMid] <= rotateArray[pEnd]){
                pEnd = pMid;
            }
        }

        return rotateArray[pEnd];
    }

    int searchMin(vector<int>& rotateArray){

        int min = rotateArray[0];
        for(int i = 0; i < rotateArray.size(); i++){
            if(rotateArray[i] < min){
                min = rotateArray[i];
            }
        }
        return min;
    }
};

    12：矩阵中的路径

    class Solution {
public:
    bool hasPath(char* matrix, int rows, int cols, char* str)
    {
        if(matrix == nullptr || rows <= 0 || cols <= 0 || str == nullptr)
            return false;

        vector<vector<bool>> visited;
        for(int i = 0; i < rows; i++){
            vector<bool> v;
            for(int j = 0; j < cols; j++){
                v.push_back(false);
            }
            visited.push_back(v);
        }

        for(int i = 0; i < rows; i++){
            for(int j = 0; j < cols; j++){
                if(hasPathCore(matrix, visited, rows, cols, i, j, str)){
                    return true;
                }
            }
        }
        return false;
    }

    bool hasPathCore(char* matrix, vector<vector<bool>>& visited, int rows, int cols, int row, int col, char* str){
        if(matrix == nullptr || rows <= 0 || cols <= 0 || str == nullptr)
            return false;

        if(*str == '\0'){
            return true;
        }

        if(col < 0 || row < 0 || col >= cols || row >= rows){
            return false;
        }

        bool res = false;
        if(matrix[row*cols+col] == *str && !visited[row][col]){
            visited[row][col] = true;
            int i = row; int j = col;
            res = hasPathCore(matrix, visited, rows, cols, i+1, j, str+1) ||
                hasPathCore(matrix, visited, rows, cols, i-1, j, str+1) ||
                hasPathCore(matrix, visited, rows, cols, i, j+1, str+1) ||
                hasPathCore(matrix, visited, rows, cols, i, j-1, str+1);

            visited[row][col] = false;
        }

        return res;
    }
};
