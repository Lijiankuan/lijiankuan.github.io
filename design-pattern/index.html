    观察者模式

    观察者模式中有两个重要概念，Observer 和 Subject。观察者模式的本质是 Subject 对象发生变化的时候，通知所有观察了这个 Subject 的 Observer，而 Subject 不需要 Observer 实际类型信息。

    class IObserver
{
public:
    virtual void OnNotify() = 0;
};

class ISubject
{
private:
    std::vector<IObserver> observers_;

public:
    // 利用这个方法可以将Observer对象注册到Subject对象中
    void RegisterObserver(IObserver observer)
    {
        observers_.Add(observer);
    }

protected:
    // 当类对象状态发生改变时候，NotifyObserver 方法通知所有注册进（调用 RegisterObserver）这个类对象的观察者
    void NotifyObserver()
    {
        foreach (auto observer: observers_)
            observer.OnNotify();
    }
};


// 新创建的类需要继承的基类
class ImportantBaseClass { /* Members */ };
// Donot-repeat-yourself 原则，Subject 作为基类，用到多重继承
class SubjectSubClass: public ImportantBaseClass, public ISubject { /* Members */ };

// 新创建的类需要继承的基类
class AnotherImportantBaseClass {...};
// 使用多重继承。若不继承 IObserve，而在 SubClass 添加 OnNotify 方法，则无法使用 Subject 的 Register 方法
class ObserverSubClass: public AnotherImportantBaseClass, public IObserve {...};
